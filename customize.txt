#!/bin/bash

echo First, we\'ll add autocomplete \(my favorite feature\). Installing via oh-my-zsh...
#git clone git://github.com/zsh-users/zsh-autosuggestions $ZSH_CUSTOM/plugins/zsh-autosuggestions

echo Adding zsh-autosuggestions to the plugins line of .zshrc...
sed 's/plugins=(/&zsh-autosuggestions /' ~/.zshrc | tee ~/.zshrc

echo Now we\'ll add syntax highlighting...
git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting
sed 's/plugins=(/&zsh-syntax-highlighting /' ~/.zshrc | tee ~/.zshrc
source ~/.zshrc

echo Now we\'ll change the theme \(it won\'t update until you restart\)...
# sed s/ZSH_THEME=\"robbyrussell\"/ZSH_THEME=\"agnoster\"/ ~/.zshrc | tee ~/.zshrc
# echo ZSH_THEME=\"agnoster\" >> ~/.zshrc

echo Adding my favorite git shortcuts a aliases...
# display to user
echo alias ga="git add ."
echo alias gcm="git checkout master"
echo alias gp="git push"
echo alias gs="git status"
echo alias add="git add ."
echo alias master="git checkout master"
echo alias push="git push"
echo alias pull="git pull"
echo alias status="git status"

# actually add to .zshrc
echo alias ga=\"git add .\" >> ~/.zshrc
echo alias gcm=\"git checkout master\" >> ~/.zshrc
echo alias gp=\"git push\" >> ~/.zshrc
echo alias gs=\"git status\" >> ~/.zshrc
echo alias add=\"git add .\" >> ~/.zshrc
echo alias master=\"git checkout master\" >> ~/.zshrc
echo alias push=\"git push\" >> ~/.zshrc
echo alias pull=\"git pull\" >> ~/.zshrc
echo alias status=\"git status\" >> ~/.zshrc

echo "ZSH_THEME=\"agnoster\"" >> ~/.zshrc

echo If you use rbenv don\'t forget to add the following to your .zshrc...

echo "export PATH=\"\$HOME/.rbenv/bin:\$PATH\""
echo eval \"\$\(rbenv init -\)\"

echo We need powerline fonts as well, downloading...
# clone
git clone https://github.com/powerline/fonts.git --depth=1
# install
cd fonts
./install.sh
# clean-up a bit
cd ..
rm -rf fonts
